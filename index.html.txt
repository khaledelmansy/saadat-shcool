        // استيراد وقراءة ملفات الإكسل (Excel) المرفوعة
        function importFromExcel(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result);
                    const workbook = XLSX.read(data, { type: 'array' });
                    
                    const firstSheetName = workbook.SheetNames[0];
                    const worksheet = workbook.Sheets[firstSheetName];
                    const rawRows = XLSX.utils.sheet_to_json(worksheet, { defval: "", raw: false });

                    if (rawRows.length === 0) {
                        showToast("لم يتم العثور على أي بيانات مسجلة داخل شيت الإكسل المرفوع.", "error");
                        return;
                    }

                    const requiredColumns = [
                        "الاسم رباعي", "الرقم القومي (14 رقم)", "رقم الجلوس", 
                        "الصف الدراسي", "اللغة العربية (من 80)", "اللغة الإنجليزية (من 60)"
                    ];

                    const headers = Object.keys(rawRows[0]);
                    const hasRequiredCols = requiredColumns.every(col => headers.includes(col));

                    if (!hasRequiredCols) {
                        showToast("تنسيق أعمدة شيت الإكسل المرفوع غير مطابق للمواصفات والقالب المطلوب.", "error");
                        return;
                    }

                    let importedStudents = [];
                    let parseErrors = 0;

                    rawRows.forEach(row => {
                        let nationalId = String(row["الرقم القومي (14 رقم)"] || "").trim();
                        // فك الأرقام المشوهة أو المكتوبة بالصيغة العلمية
                        if (nationalId.includes('.') || nationalId.includes('E') || nationalId.includes('e')) {
                            parseErrors++;
                            return;
                        }

                        if (nationalId.length !== 14 || isNaN(nationalId)) {
                            parseErrors++;
                            return;
                        }

                        const studentObj = {
                            name: String(row["الاسم رباعي"] || "").trim(),
                            nationalId: nationalId,
                            seatNo: String(row["رقم الجلوس"] || "").trim(),
                            grade: String(row["الصف الدراسي"] || "الصف الأول الثانوي").trim(),
                            militaryEval: String(row["تقييم التربية العسكرية"] || "منضبط ومجتاز بكفاءة عالية (90/100)").trim(),
                            grades: {
                                arabic: Math.min(80, Math.max(0, parseInt(row["اللغة العربية (من 80)"]) || 0)),
                                english: Math.min(60, Math.max(0, parseInt(row["اللغة الإنجليزية (من 60)"]) || 0)),
                                math: Math.min(80, Math.max(0, parseInt(row["الرياضيات (من 80)"]) || 0)),
                                physics: Math.min(40, Math.max(0, parseInt(row["الفيزياء (من 40)"]) || 0)),
                                chemistry: Math.min(40, Math.max(0, parseInt(row["الكيمياء (من 40)"]) || 0)),
                                biology: Math.min(40, Math.max(0, parseInt(row["الأحياء (من 40)"]) || 0))
                            }
                        };

                        importedStudents.push(studentObj);
                    });

                    if (importedStudents.length === 0) {
                        showToast("لم ينجح استيراد أي طالب، تأكد من سلامة الأرقام القومية.", "error");
                        return;
                    }

                    students = importedStudents;
                    localStorage.setItem('sadat_military_students', JSON.stringify(students));
                    renderAdminStudentsTable();
                    
                    if (parseErrors > 0) {
                        showToast(`تم استيراد ${importedStudents.length} طالب بنجاح، وتخطي ${parseErrors} سطر لعدم صلاحية الرقم القومي.`, "info");
                    } else {
                        showToast(`تم استيراد وتحديث كشوفات كافة الطلاب المسجلين بالملف بنجاح (${importedStudents.length} طالب).`, "success");
                    }

                } catch (err) {
                    console.error(err);
                    showToast("حدث خطأ غير متوقع أثناء معالجة ملف الإكسل.", "error");
                }
                document.getElementById('excelFileInput').value = '';
            };
            reader.readAsArrayBuffer(file);
        }

        // إظهار تنبيه وإشعار مخصص وأنيق للمستخدم
        function showToast(message, type = 'success') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            toast.className = `fade-in p-4 rounded-xl shadow-lg border text-sm flex items-center gap-3 bg-white pointer-events-auto transition duration-300`;
            
            let icon = '';
            if (type === 'success') {
                toast.classList.add('border-emerald-500', 'text-emerald-950');
                icon = '<i class="fa-solid fa-circle-check text-emerald-600 text-lg"></i>';
            } else if (type === 'error') {
                toast.classList.add('border-red-500', 'text-red-950');
                icon = '<i class="fa-solid fa-circle-xmark text-red-600 text-lg"></i>';
            } else {
                toast.classList.add('border-blue-500', 'text-blue-950');
                icon = '<i class="fa-solid fa-circle-info text-blue-600 text-lg"></i>';
            }

            toast.innerHTML = `
                ${icon}
                <div class="flex-1 font-semibold">${message}</div>
                <button onclick="this.parentElement.remove()" class="text-gray-400 hover:text-gray-600"><i class="fa-solid fa-xmark"></i></button>
            `;

            container.appendChild(toast);
            setTimeout(() => {
                toast.style.opacity = '0';
                setTimeout(() => toast.remove(), 300);
            }, 5000);
        }

        // دالة مخصصة لطلب تأكيد الإجراءات واستبدال confirm الممنوعة
        function askConfirm(message, callback) {
            const modal = document.getElementById('custom-confirm-modal');
            const msgText = document.getElementById('confirm-modal-text');
            const confirmBtn = document.getElementById('confirm-modal-yes');
            const cancelBtn = document.getElementById('confirm-modal-no');

            msgText.textContent = message;
            modal.classList.remove('hidden');
            modal.classList.add('flex');

            const nextConfirm = confirmBtn.cloneNode(true);
            const nextCancel = cancelBtn.cloneNode(true);
            confirmBtn.parentNode.replaceChild(nextConfirm, confirmBtn);
            cancelBtn.parentNode.replaceChild(nextCancel, cancelBtn);

            nextConfirm.addEventListener('click', () => {
                modal.classList.add('hidden');
                modal.classList.remove('flex');
                callback();
            });

            nextCancel.addEventListener('click', () => {
                modal.classList.add('hidden');
                modal.classList.remove('flex');
            });
        }

        // استعلام وبحث الطلاب عن النتائج عبر الرقم القومي
        function handleSearch(event) {
            event.preventDefault();
            const searchInput = document.getElementById('nationalIdInput').value.trim();
            const resultArea = document.getElementById('result-display-area');

            if (!searchInput) {
                showToast("من فضلك أدخل الرقم القومي للطالب للبدء في البحث.", "error");
                return;
            }

            if (searchInput.length !== 14 || isNaN(searchInput)) {
                showToast("الرقم القومي غير صالح، يرجى كتابة 14 رقماً صحيحاً تماماً.", "error");
                return;
            }

            const foundStudent = students.find(student => student.nationalId === searchInput);

            if (foundStudent) {
                showToast("تم العثور على النتيجة الرسمية للطالب بنجاح! جاري عرض الشهادة.", "success");
                displayStudentCertificate(foundStudent);
                
                setTimeout(() => {
                    resultArea.classList.remove('hidden');
                    resultArea.scrollIntoView({ behavior: 'smooth' });
                }, 200);
            } else {
                showToast("عذراً، الرقم القومي الذي أدخلته غير مسجل حالياً بكشوف النتائج.", "error");
                resultArea.classList.add('hidden');
            }
        }

        function clearSearch() {

            document.getElementById('search-section').scrollIntoView({ behavior: 'smooth' });
        }

        // صياغة وإعداد وثيقة بيان الدرجات وبطاقات الشهادات للطباعة
        function displayStudentCertificate(student) {
            document.getElementById('cert-student-name').textContent = student.name;
            document.getElementById('cert-student-grade').textContent = student.grade;
            document.getElementById('cert-seat-no').textContent = student.seatNo;
            document.getElementById('cert-national-id').textContent = student.nationalId;
            document.getElementById('cert-military-eval').textContent = student.militaryEval;

            const today = new Date();
            document.getElementById('cert-current-date').textContent = today.toLocaleDateString('ar-EG');

            const tableBody = document.getElementById('cert-grades-table-body');
            tableBody.innerHTML = '';

            let studentTotal = 0;
            let examMaxTotal = 0;
            let hasFailedSubject = false;

            for (const key in subjectsConfig) {
                const config = subjectsConfig[key];
                const studentGrade = student.grades[key] !== undefined ? student.grades[key] : 0;
                
                studentTotal += studentGrade;
                examMaxTotal += config.max;

                const isFailed = studentGrade < config.min;
                if (isFailed) {
                    hasFailedSubject = true;
                }

                const percentage = (studentGrade / config.max) * 100;
                let subjectEvaluation = '';
                let textClass = 'text-gray-800';

                if (isFailed) {
                    subjectEvaluation = 'دون النهاية الصغرى (راسب)';
                    textClass = 'text-red-600 font-bold';
                } else if (percentage >= 85) {
                    subjectEvaluation = 'ممتاز';
                    textClass = 'text-emerald-700 font-bold';
                } else if (percentage >= 75) {
                    subjectEvaluation = 'جيد جداً';
                    textClass = 'text-emerald-600';
                } else if (percentage >= 65) {
                    subjectEvaluation = 'جيد';
                    textClass = 'text-blue-600';
                } else if (percentage >= 50) {
                    subjectEvaluation = 'مقبول';
                    textClass = 'text-yellow-600';
                }

                const row = document.createElement('tr');
                row.className = isFailed ? 'bg-red-50/50' : 'hover:bg-gray-50';
                row.innerHTML = `
                    <td class="border border-gray-200 p-2 text-right font-medium text-military-900">${config.name}</td>
                    <td class="border border-gray-200 p-2 font-mono text-gray-500">${config.max}</td>
                    <td class="border border-gray-200 p-2 font-mono text-gray-500">${config.min}</td>
                    <td class="border border-gray-200 p-2 font-mono ${textClass}">${studentGrade}</td>
                    <td class="border border-gray-200 p-2 text-xs font-semibold ${textClass}">${subjectEvaluation}</td>
                `;
                tableBody.appendChild(row);
            }

            document.getElementById('cert-total-score').textContent = studentTotal;
            document.getElementById('cert-max-score').textContent = examMaxTotal;

            const totalPercentage = ((studentTotal / examMaxTotal) * 100).toFixed(1);
            document.getElementById('cert-percentage').textContent = `%${totalPercentage}`;

            const certStatusBadge = document.getElementById('cert-status-badge');
            let finalGradeText = '';

            if (hasFailedSubject) {
                certStatusBadge.textContent = "راسب وله دور ثانٍ";
                certStatusBadge.className = "bg-red-100 text-red-800 font-bold px-2.5 py-1 rounded text-xs";
                finalGradeText = "ضعيف / دور ثانٍ";
                document.getElementById('cert-final-grade').className = "text-2xl font-black mt-1 text-red-600";
            } else {
                certStatusBadge.textContent = "ناجح ومجتاز";
                certStatusBadge.className = "bg-emerald-100 text-emerald-800 font-bold px-2.5 py-1 rounded text-xs";
                
                if (totalPercentage >= 85) {
                    finalGradeText = "ممتاز";
                } else if (totalPercentage >= 75) {
                    finalGradeText = "جيد جداً";
                } else if (totalPercentage >= 65) {
                    finalGradeText = "جيد";
                } else {
                    finalGradeText = "مقبول";
                }
                document.getElementById('cert-final-grade').className = "text-2xl font-black mt-1 text-emerald-700";
            }

            document.getElementById('cert-final-grade').textContent = finalGradeText;
        }

        // تعبئة وجدول الطلاب في واجهة الأدمن
        function renderAdminStudentsTable() {
            const tbody = document.getElementById('admin-students-table-body');
            tbody.innerHTML = '';

            students.forEach((student, index) => {
                let studentSum = 0;
                let maxPossible = 0;
                for (const key in subjectsConfig) {
                    studentSum += student.grades[key] || 0;
                    maxPossible += subjectsConfig[key].max;
                }
                const percentage = ((studentSum / maxPossible) * 100).toFixed(1);

                const row = document.createElement('tr');
                row.className = "border-b border-military-800 hover:bg-military-800/60";
                row.innerHTML = `
                    <td class="p-2 font-mono text-gray-300 font-bold">${student.seatNo}</td>
                    <td class="p-2 font-semibold text-white">${student.name}</td>
                    <td class="p-2 font-mono text-yellow-400 font-semibold select-all">${student.nationalId}</td>
                    <td class="p-2 text-gray-300">${student.grade}</td>
                    <td class="p-2 text-emerald-400 font-bold font-mono">${percentage}%</td>
                    <td class="p-2 text-center flex justify-center gap-2">
                        <button onclick="editStudent(${index})" class="bg-yellow-500 hover:bg-yellow-600 text-military-950 px-2 py-1 rounded text-[11px] font-bold">
                            تعديل
                        </button>
                        <button onclick="deleteStudentPrompt(${index})" class="bg-red-700 hover:bg-red-800 text-white px-2 py-1 rounded text-[11px] font-bold">
                            حذف
                        </button>
                    </td>
                `;
                tbody.appendChild(row);
            });
        }

        // حفظ بيانات الطالب الجديد أو المحدث وتخزينها
        function saveStudent(event) {
            event.preventDefault();
            
            const index = parseInt(document.getElementById('edit-index').value);
            const nationalId = document.getElementById('form-national-id').value.trim();

            if (nationalId.length !== 14 || isNaN(nationalId)) {
                showToast("يجب أن يتكون الرقم القومي للطلاب من 14 رقماً صحيحاً تماماً للتسجيل والمطابقة.", "error");
                return;
            }

            const updatedStudent = {
                name: document.getElementById('form-name').value.trim(),
                nationalId: nationalId,
                seatNo: document.getElementById('form-seat').value.trim(),
                grade: document.getElementById('form-grade').value,
                militaryEval: document.getElementById('form-military-eval').value,
                grades: {
                    arabic: parseInt(document.getElementById('grade-arabic').value),
                    english: parseInt(document.getElementById('grade-english').value),
                    math: parseInt(document.getElementById('grade-math').value),
                    physics: parseInt(document.getElementById('grade-physics').value),
                    chemistry: parseInt(document.getElementById('grade-chemistry').value),
                    biology: parseInt(document.getElementById('grade-biology').value)
                }
            };

            if (index === -1) {
                // التحقق من تكرار الرقم القومي
                if (students.some(s => s.nationalId === nationalId)) {
                    showToast("هذا الرقم القومي مستخدم ومسجل مسبقاً لطالب آخر بالدفعة العسكرية.", "error");
                    return;
                }
                students.push(updatedStudent);
                showToast(`تم إدراج الطالب البطل "${updatedStudent.name}" إلى قاعدة البيانات بنجاح!`, "success");
            } else {
                // التأكد من عدم سرقة رقم قومي لطالب ثالث
                const duplicated = students.some((s, i) => s.nationalId === nationalId && i !== index);
                if (duplicated) {
                    showToast("الرقم القومي المدخل يخص طالباً آخر مسجلاً في القائمة.", "error");
                    return;
                }
                students[index] = updatedStudent;
                showToast(`تم تحديث بيانات درجات البطل "${updatedStudent.name}" بنجاح في السجلات الإلكترونية.`, "success");
            }

            localStorage.setItem('sadat_military_students', JSON.stringify(students));
            renderAdminStudentsTable();
            resetForm();
        }

        function resetForm() {
            document.getElementById('edit-index').value = "-1";
            document.getElementById('student-form').reset();
        }
